# 注册插件路由

  • 函数 this.injectRouter(routers)
    let routers = [
  ['/calc/start', 'calc/start', 'index'],
  // ['路由名', '控制器', 'Action方法'],
]
this.injectRouter(routers) // 执行方法注入路由
  • 注入路由会自动添加/plg/${插件ID}的前缀
  • 控制器路径相对于controller目录下
  • Action方法 方法为index时可以省略

# 启动算法

  1. 释放插件到指定的位置
  2. 获取插件容器状态
    • 获取状态
    // 容器名称 统一为 civi_ 加上 插件ID
let docker_name = 'civi_' + this.id

// 发送docker 指令到宿主机查看容器状态
let docker_status = await civi.command.execSync('docker ps -a --format "{{.Names}}@{{.Status}}"')
let status_arr = docker_status.split('\n')
let appStatus = -1
// 解析结果字符串 判断容器状态
status_arr = status_arr.filter(item => item).forEach(item=> {
  if (item.split('@')[0] === docker_name) {
    appStatus = 0
    if (item.split('@')[1].split(' ')[0] === 'Up') appStatus = 1
  }
})
    • 容器不存在，新建容器
    if (appStatus === -1) {
  await civi.command.execSync(`docker run -it -d --gpus all --restart=always -v /civi/test:/civi --net=host --ipc=host --name=${docker_name} civi:swzc /bin/bash -c "cd /civi/algs/td && sh start.sh td"`)
}
    • 容器存在，但未运行，启动容器
    if (appStatus === 0) {
  await civi.command.execSync(`docker stop ${docker_name} && docker start ${docker_name}`)
}

注意

插件存在算法时需要先注册路由和数据模型之后再启动算法
