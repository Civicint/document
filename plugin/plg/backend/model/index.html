# 数据模型

数据模型放在插件根目录下的model 文件夹下，初始化时数据表会自动添加当前插件ID的前缀

# 代码

const Schema = civi.DB.Schema;

module.exports = class extends civi.DB {
  get schema() {
    var _schema = new Schema({
      name: {type: String, required: true}, // 数据模型字段
    });
    return _schema;
  }
};

# 使用

# 在Plugin类中使用

// 不用带上插件前缀
this.db('name')

# 在全局使用

// 方式一
civi.db('${插件ID}_${表名称}')

// 方式二
civi.plugins['${插件ID}'].db('${表名称}')

# 在BaseController类中使用

// 不用带上插件前缀
this.plg.db('name')
