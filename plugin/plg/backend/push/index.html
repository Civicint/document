# 推流控制器

module.exports = class extends civi.BaseController {
  isNoAuth = true // 设置当前控制器不经过登录验证
  async indexAction () {
    let data = this.post()

    /**
     * 算法的一些操作
     **/

    // 以下代码必须存在 把推流数据据绑定到全局 统一推送给客户端
    civi.pullWsData[data.type] = civi.pullWsData[data.type] || {}
    civi.pullWsData[data.type][data.cam] = data
  }
}
