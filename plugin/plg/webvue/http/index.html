# HTTP请求

# get请求

  • this.api.GET(url, params)
  • url 请求路径
  • params URL参数
await this.api.GET('/api/001', {a: 1})

# post请求

  • this.api.POST(url, data)
  • url 请求路径
  • data 请求参数
await this.api.POST('/api/001', {a: 1})

# request请求

  • this.api.REQUEST(option)
  • option axios的请求配置
await this.api.REQUEST({
  url: '/api/001',
  method: 'post',
  headers: { "Content-Type": "application/json" }
})
