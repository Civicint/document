# 入口文件

VUE主框架引入后自动执行Vue.use() 注册插件

# 基础代码

// 插件安装方法
const install = function (Vue) {
  if (install.installed) return;
};

// 导出插件
export default { install };

# 路由注入

import BaseIndex from "./pages/baseIndex/baseIndex.vue";

const install = function (Vue) {
  if (install.installed) return;

  // 获取主框架提供的对象
  const { $pages } = Vue.prototype;

  // 注入路由
  let route = [
    {
      path: "/index",
      meta: { title: "通道点数" },
      component: BaseIndex,
    },
  ];

  // 添加路由
  route = $pages.addPage("${插件ID}", route);
};

$pages.addPage 方法会给路由添加 /plg/${插件ID} 的前缀，并返回完整路由

# 右侧菜单注入

const install = function (Vue) {
  if (install.installed) return;

  // 获取主框架提供的对象
  const { $menus } = Vue.prototype;

  // 注入菜单
  let menu = {
    path: '/plg/td/index', // 菜单跳转路径
    title: route[0].meta.title, // 菜单标题
    img: require("./assets/icon.svg"), // 菜单图标，使用png 或者svg 带有透明的格式
  };
  $menus.addMenu(1, menu);
};
