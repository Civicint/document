# WebSocket

# 监听事件

  • this.ws.on(eventName, callback);
  • eventName 事件名称
  • callback 监听回调函数
this.ws.on("cameraPreview", (data) => {
  console.log('收到数据', data)
});

# 取消监听

  • this.ws.remove(eventName, callback);
  • eventName 事件名称
  • callback 监听回调函数
// 监听回调
function callback (data) {
  console.log('收到数据', data)
}

// 监听事件
this.ws.on("cameraPreview", callback);

// 取消监听
this.ws.remove("cameraPreview", callback);

# 发送事件

  • this.ws.emit(eventName, eventData)
  • eventName 事件名称
  • eventData 事件数据
this.ws.emit("cameraPreview", {
  rtsp: 'rtsp://....'
});

# 视频预览

let eventName = 'pullVideo_' + '$插件ID' + '_' + 'rtsp://...'
this.ws.on(eventName, (data) => {
  console.log('收到数据', data)
});

监听视频预览数据的事件名称，pullVideo + $插件ID + RTSP链接 使用 - 连接的格式拼接出完整的事件名称
